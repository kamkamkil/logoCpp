cmake_minimum_required(VERSION 3.5.0)
project(worldCounter VERSION 0.1.0 LANGUAGES C CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED True)
find_package(PkgConfig)
find_package(PkgConfig REQUIRED)

pkg_check_modules(GTKMM_VARS REQUIRED IMPORTED_TARGET gtkmm-4.0)
pkg_check_modules(GTKMM_LIB REQUIRED IMPORTED_TARGET gtkmm-4.0)
add_library(interpreter interpreter.hpp interpreter.cpp)
add_library(renderer renderer.hpp renderer.cpp)
add_executable(main main.cpp)
target_link_libraries(main interpreter)
target_link_libraries(renderer PUBLIC PkgConfig::GTKMM_VARS)
target_link_libraries(renderer PUBLIC PkgConfig::GTKMM_LIB) 
target_link_libraries(interpreter PUBLIC  renderer)
