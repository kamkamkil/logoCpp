cmake_minimum_required(VERSION 3.5.0)
project(worldCounter VERSION 0.1.0 LANGUAGES C CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED True)
add_library(interpreter interpreter.cpp)
add_library(renderer renderer.cpp)
add_executable(main main.cpp)
target_link_libraries(interpreter PUBLIC  renderer)
target_link_libraries(main  PUBLIC renderer)
